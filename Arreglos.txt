fun main() {
   val notas: Array<Double?> = arrayOfNulls(5)
   
   notas[0] = 6.5
   notas[1] = 7.0
   notas[2] = 8.0 
   notas[3] = 9.5 
   notas[4] = 7.5 
    
   var suma = 0.0 
    for (i in notas.indices) {
        suma += notas[i]!!
    }
    
    val promedio = suma / notas.size
    
    println("Promedio de las notas: $promedio ")
}
